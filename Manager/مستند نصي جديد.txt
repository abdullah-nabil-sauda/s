                    entry.target.classList.add('visible');
                    scrollObserver.unobserve(entry.target);
                }
            });
        }, { threshold: 0.1 });
        document.querySelectorAll('.fade-up').forEach(el => scrollObserver.observe(el));

        // Smooth Scroll
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                if (targetId === '#') return;
                const targetElement = document.querySelector(targetId);
                if (targetElement) {
                    window.scrollTo({ top: targetElement.getBoundingClientRect().top + window.pageYOffset - 80, behavior: 'smooth' });
                }
            });
        });

        // Navbar Shadow
        const navbar = document.getElementById('navbar');
        window.addEventListener('scroll', () => {
            navbar.classList.toggle('shadow-md', window.scrollY > 10);
        }, { passive: true });

        // ====== الحل الاحترافي: ربط الشريط السفلي مع زر الواتساب ======
        const stickyCta = document.getElementById('sticky-cta-bar');
        const whatsappFloat = document.getElementById('whatsapp-float');
        let stickyCtaShown = false;

        window.addEventListener('scroll', () => {
            const threshold = window.innerHeight * 0.8;
            const footer = document.querySelector('footer');
            const footerTop = footer ? footer.getBoundingClientRect().top : Infinity;

            // منطق إظهار/إخفاء الشريط السفلي
            if (window.scrollY > threshold && footerTop > window.innerHeight) {
                if (!stickyCtaShown) {
                    stickyCta.classList.add('visible');
                    whatsappFloat.classList.add('lifted'); // رفع زر الواتساب
                    stickyCtaShown = true;
                }
            } else {
                if (stickyCtaShown) {
                    stickyCta.classList.remove('visible');
                    whatsappFloat.classList.remove('lifted'); // إرجاع زر الواتساب
                    stickyCtaShown = false;
                }
            }

        // Lightbox

            lightbox.classList.add('active');
            document.body.style.overflow = 'hidden';

        function closeLightbox() {
            lightbox.classList.remove('active');
            document.body.style.overflow = '';

            if (e.key === 'Escape' && lightbox.classList.contains('active')) closeLightbox();
        });

        // FAQ Toggle

            const icon = button.querySelector('.faq-icon');

                content.classList.remove('hidden');
                icon.classList.add('rotate-180');
            }
        }

                    button.textContent = 'نسخ';
                    button.classList.remove('copied');

            }).catch(() => {

                document.body.appendChild(textarea);
                textarea.select();
                document.execCommand('copy');
                document.body.removeChild(textarea);
                button.textContent = '✓ تم النسخ';
                button.classList.add('copied');
                setTimeout(() => {
                    button.textContent = 'نسخ';
                    button.classList.remove('copied');
                }, 2000);
